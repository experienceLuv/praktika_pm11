CREATE OR REPLACE FUNCTION keeperpro.set_user_login()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    email_prefix text;
BEGIN
    email_prefix := split_part(NEW.email, '@', 1);

    UPDATE keeperpro.users
    SET login = email_prefix || NEW.user_id
    WHERE user_id = NEW.user_id;

    RETURN NEW;
END;
$$;
